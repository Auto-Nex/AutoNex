int LED_PIN = D7;
int SENSOR_PIN = D2;
int out =  0;

void setup() {
pinMode(SENSOR_PIN,INPUT_PULLUP);
Serial.begin(115200);
}

void loop() {
if(digitalRead(SENSOR_PIN)==LOW){
  
  if(out==0){
   analogWrite(LED_PIN,85);
    Serial.println("LED_PIN 85");
    out=85;
    delay(1000);
  }else if(out==85){
    analogWrite(LED_PIN,170);
  Serial.println("LED_PIN 170");
  out=170;
   delay(1000);
}else if(out==170){
  analogWrite(LED_PIN,255);
  Serial.println("LED_PIN 255");
  out=255;
   delay(1000);
}else if(out==255){
 analogWrite(LED_PIN,0);
  Serial.println("LED_PIN 0");
  out=0;
   delay(1000);
}
}
}
